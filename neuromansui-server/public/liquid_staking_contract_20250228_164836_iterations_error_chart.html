module 0x0::my_module {

    use std::vector;
    use std::signer;
    use sui::coin::{Coin, destroy, value, create};

    /// Error codes used in the module.
    const E_LOCKED: u64 = 1;
    const E_UNAUTHORIZED: u64 = 2;

    /// A resource that represents the global staking pool.
    ///
    /// This pool aggregates staked SUI, mints liquid staking tokens,
    /// and manages reward accrual and delegation to validators.
    struct StakingPool has key {
        /// The admin address that controls certain privileged operations.
        admin: address,
        /// Total amount of SUI staked in the pool.
        total_staked: u64,
        /// Total supply of liquid staking tokens outstanding.
        total_liquid_tokens: u64,
        /// Accumulated rewards available for distribution.
        pending_rewards: u64,
        /// A vector tracking delegation amounts to validators.
        /// Each item is a tuple (validator_address, delegated_amount).
        validator_delegations: vector<(address, u64)>,
    }

    /// A resource representing a user's deposit ticket.
    ///
    /// This ticket is issued when a user stakes SUI and represents
    /// their position in the pool with associated liquid tokens.
    /// A non-zero lock_until value indicates that the deposit is locked
    /// until the specified unix timestamp.
    struct LiquidTicket has key, store {
        owner: address,
        liquid_amount: u64,
        lock_until: u64,
    }

    ///////////////////////////
    // Module Initialization //
    ///////////////////////////

    /// Initializes the staking pool.
    ///
    /// The caller becomes the admin and owner of the StakingPool resource.
    public fun initialize(admin: &signer) {
        let pool = StakingPool {
            admin: signer::address_of(admin),
            total_staked: 0,
            total_liquid_tokens: 0,
            pending_rewards: 0,
            validator_delegations: vector::empty(),
        };
        move_to(admin, pool);
    }

    //////////////////////
    // User Interfaces  //
    //////////////////////

    /// Stake SUI into the pool.
    ///
    /// Accepts a Coin<SUI> (native SUI coin) from the user, mints liquid staking tokens
    /// represented by a LiquidTicket, and updates the staking pool state.
    /// The user can optionally specify a lock period via a unix timestamp (lock_until).
    /// The chosen validator for delegation is also recorded.
    public fun stake(
        pool_addr: address,
        user: &signer,
        sui_coin: Coin,
        lock_until: u64,
        validator: address
    ): LiquidTicket {
        // Borrow the staking pool resource published at pool_addr.
        let pool_ref = borrow_global_mut<StakingPool>(pool_addr);

        // Retrieve the coin value.
        let amount = value(&sui_coin);

        // Determine the number of liquid tokens to mint.
        // If the pool is empty, mint tokens at a 1:1 ratio.
        // Otherwise, mint tokens proportional to the user's contribution.
        let minted_liquid: u64;
        if (pool_ref.total_staked == 0) {
            minted_liquid = amount;
        } else {
            minted_liquid = amount * pool_ref.total_liquid_tokens / pool_ref.total_staked;
        }

        // Update pool totals.
        pool_ref.total_staked = pool_ref.total_staked + amount;
        pool_ref.total_liquid_tokens = pool_ref.total_liquid_tokens + minted_liquid;

        // Update validator delegation.
        let idx = index_of_validator(&pool_ref.validator_delegations, validator);
        if (idx < vector::length(&pool_ref.validator_delegations)) {
            let (val, delegated_amount) = *vector::borrow_mut(&mut pool_ref.validator_delegations, idx);
            vector::borrow_mut(&mut pool_ref.validator_delegations, idx).1 = delegated_amount + amount;
        } else {
            vector::push_back(&mut pool_ref.validator_delegations, (validator, amount));
        }

        // Destroy the SUI coin as it is now added to the pool.
        destroy(sui_coin);

        // Mint a LiquidTicket representing the user's staked position.
        let ticket = LiquidTicket {
            owner: signer::address_of(user),
            liquid_amount: minted_liquid,
            lock_until: lock_until,
        };

        ticket
    }

    /// Unstake SUI from the pool.
    ///
    /// Consumes the user's LiquidTicket and returns a Coin<SUI> representing the
    /// underlying staked amount plus rewards. Enforces any lock period specified during staking.
    public fun unstake(
        pool_addr: address,
        ticket: LiquidTicket,
        current_time: u64
    ): Coin {
        // Ensure the deposit is unlocked.
        if (ticket.lock_until > current_time) {
            abort E_LOCKED;
        }

        let pool_ref = borrow_global_mut<StakingPool>(pool_addr);

        // Calculate total pool value including rewards.
        let total_value = pool_ref.total_staked + pool_ref.pending_rewards;

        // User's share is proportional to their liquid token holdings.
        let user_value = ticket.liquid_amount * total_value / pool_ref.total_liquid_tokens;

        // Update pool totals proportionally.
        let staked_reduction = ticket.liquid_amount * pool_ref.total_staked / pool_ref.total_liquid_tokens;
        let rewards_reduction = ticket.liquid_amount * pool_ref.pending_rewards / pool_ref.total_liquid_tokens;
        pool_ref.total_staked = pool_ref.total_staked - staked_reduction;
        pool_ref.pending_rewards = pool_ref.pending_rewards - rewards_reduction;
        pool_ref.total_liquid_tokens = pool_ref.total_liquid_tokens - ticket.liquid_amount;

        // Return a Coin<SUI> representing the unstaked value.
        let unstaked_coin = create(user_value);

        // The LiquidTicket resource is consumed (dropped) here.
        unstaked_coin
    }

    //////////////////////
    // Admin Interfaces //
    //////////////////////

    /// Deposit rewards into the staking pool.
    ///
    /// Only the admin can call this function.
    /// Rewards are added to the pool's pending_rewards for distribution.
    public fun deposit_rewards(
        pool_addr: address,
        admin: &signer,
        rewards: Coin
    ) {
        let pool_ref = borrow_global_mut<StakingPool>(pool_addr);
        if (signer::address_of(admin) != pool_ref.admin) {
            abort E_UNAUTHORIZED;
        };
        let reward_amount = value(&rewards);
        pool_ref.pending_rewards = pool_ref.pending_rewards + reward_amount;
        destroy(rewards);
    }

    /// Delegate additional SUI to a validator.
    ///
    /// Only the admin can modify the delegation allocations.
    public fun delegate(
        pool_addr: address,
        admin: &signer,
        validator: address,
        amount: u64
    ) {
        let pool_ref = borrow_global_mut<StakingPool>(pool_addr);
        if (signer::address_of(admin) != pool_ref.admin) {
            abort E_UNAUTHORIZED;
        };
        let idx = index_of_validator(&pool_ref.validator_delegations, validator);
        if (idx < vector::length(&pool_ref.validator_delegations)) {
            let (val, delegated_amount) = *vector::borrow_mut(&mut pool_ref.validator_delegations, idx);
            vector::borrow_mut(&mut pool_ref.validator_delegations, idx).1 = delegated_amount + amount;
        } else {
            vector::push_back(&mut pool_ref.validator_delegations, (validator, amount));
        }
    }

    ///////////////////////////
    // Internal Helper Funcs //
    ///////////////////////////

    /// Returns the index of the validator in the delegation vector.
    /// If the validator is not found, returns the length of the vector.
    fun index_of_validator(delegations: &vector<(address, u64)>, validator: address): u64 {
        let len = vector::length(delegations);
        let mut i = 0;
        while (i < len) {
            let (val, _) = *vector::borrow(delegations, i);
            if (val == validator) {
                return i;
            };
            i = i + 1;
        };
        i
    }
}
