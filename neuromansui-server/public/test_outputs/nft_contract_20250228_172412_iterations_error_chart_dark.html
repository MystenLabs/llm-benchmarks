module 0x0::my_module {
    use std::signer;
    use sui::object::{UID, new_uid};
    use sui::tx_context::TxContext;

    // Resource representing an NFT with metadata.
    resource struct NFT has key {
        id: UID,                  // Unique object identifier.
        owner: address,           // Owner's address.
        name: vector<u8>,         // NFT name.
        description: vector<u8>,  // NFT description.
        image_url: vector<u8>     // URL pointing to the NFT image.
    }

    // MintCap resource provides minting authority.
    resource struct MintCap has key {
        admin: address,           // Administrator allowed to mint NFTs.
    }

    ////////////////////////////////////////////////////////////////////////////
    // Module Initialization
    ////////////////////////////////////////////////////////////////////////////

    // Initializes the module by creating a minting capability.
    // This function should be called only once by the administrator.
    public entry fun init(admin: &signer, _ctx: &mut TxContext): MintCap {
        let admin_addr = signer::address_of(admin);
        MintCap { admin: admin_addr }
    }

    ////////////////////////////////////////////////////////////////////////////
    // NFT Minting
    ////////////////////////////////////////////////////////////////////////////

    // Mints a new NFT with provided metadata.
    // Can only be executed by the admin holding the MintCap.
    public entry fun mint_nft(
        admin: &signer,
        mint_cap: MintCap,
        recipient: address,
        name: vector<u8>,
        description: vector<u8>,
        image_url: vector<u8>,
        ctx: &mut TxContext
    ): (NFT, MintCap) {
        let admin_addr = signer::address_of(admin);
        // Enforce access control: Only the admin specified in MintCap can mint.
        assert!(admin_addr == mint_cap.admin, 100);
        // Generate a unique identifier for the NFT.
        let new_id = new_uid(ctx);
        // Create the NFT resource with specified metadata.
        let nft = NFT {
            id: new_id,
            owner: recipient,
            name,
            description,
            image_url
        };
        (nft, mint_cap)
    }

    ////////////////////////////////////////////////////////////////////////////
    // NFT Transfer
    ////////////////////////////////////////////////////////////////////////////

    // Transfers an NFT to a new owner.
    // Only the current owner (signer) can initiate the transfer.
    public entry fun transfer_nft(sender: &signer, nft: NFT, new_owner: address): NFT {
        let sender_addr = signer::address_of(sender);
        // Ensure that only the owner of the NFT can transfer it.
        assert!(sender_addr == nft.owner, 101);
        // Re-create the NFT with the updated owner while keeping the metadata and id unchanged.
        NFT {
            id: nft.id,
            owner: new_owner,
            name: nft.name,
            description: nft.description,
            image_url: nft.image_url
        }
    }

    ////////////////////////////////////////////////////////////////////////////
    // NFT Metadata Update
    ////////////////////////////////////////////////////////////////////////////

    // Updates the NFT metadata.
    // Only the owner of the NFT (signer) is allowed to perform metadata updates.
    public entry fun update_nft_metadata(
        sender: &signer,
        nft: NFT,
        new_name: vector<u8>,
        new_description: vector<u8>,
        new_image_url: vector<u8>
    ): NFT {
        let sender_addr = signer::address_of(sender);
        // Verify that the caller is the owner of the NFT.
        assert!(sender_addr == nft.owner, 102);
        // Re-create the NFT with updated metadata and the same owner and id.
        NFT {
            id: nft.id,
            owner: nft.owner,
            name: new_name,
            description: new_description,
            image_url: new_image_url
        }
    }
}

module 0x0::my_module {
    // Import the built‐in signer module from 0x1.
    use 0x1::signer;
    // Import the UID type for Sui objects.
    use sui::object::UID;
    // Import the transaction context module.
    use sui::tx_context::TxContext;

    ////////////////////////////////////////////////////////////////////////////
    // Resource Declarations
    ////////////////////////////////////////////////////////////////////////////

    // The NFT resource represents a non-fungible token with metadata.
    // NOTE: Sui objects must have an 'id: UID' as the first field.
    // Abilities 'key', 'store', and 'drop' are required for objects.
    public struct NFT has key, store, drop {
        id: UID,                  // Unique object identifier (must be the first field).
        owner: address,           // Owner's address.
        name: vector<u8>,         // NFT name.
        description: vector<u8>,  // NFT description.
        image_url: vector<u8>     // URL pointing to the NFT image.
    }

    // The MintCap resource grants minting authority.
    // It is structured as a Sui object and must include an id field as its first field.
    public struct MintCap has key, store, drop {
        id: UID,       // Unique identifier for the minting capability.
        admin: address // The administrator authorized to mint NFTs.
    }

    ////////////////////////////////////////////////////////////////////////////
    // Initialization Function
    ////////////////////////////////////////////////////////////////////////////

    // Initializes the contract by creating a minting capability.
    // This function is not marked as an entry function because initialization
    // should be performed as part of a deployment transaction.
    public fun init(admin: &signer, ctx: &mut TxContext): MintCap {
        let admin_addr = signer::address_of(admin);
        // Generate a new UID for the MintCap.
        let cap_id = UID::new(ctx);
        MintCap { id: cap_id, admin: admin_addr }
    }

    ////////////////////////////////////////////////////////////////////////////
    // NFT Minting
    ////////////////////////////////////////////////////////////////////////////

    // Mints a new NFT with the provided metadata.
    // Access control: Only the admin specified in the MintCap (retrieved via TxContext)
    // is allowed to mint an NFT.
    public entry fun mint_nft(
        mint_cap: MintCap,
        recipient: address,
        name: vector<u8>,
        description: vector<u8>,
        image_url: vector<u8>,
        ctx: &mut TxContext
    ): (NFT, MintCap) {
        // Retrieve the sender's address from the transaction context.
        let sender = TxContext::sender(ctx);
        // Enforce access control: only the admin can mint.
        assert!(sender == mint_cap.admin, 100);
        // Generate a new UID for the NFT.
        let new_id = UID::new(ctx);
        // Create the NFT with the given metadata.
        let nft = NFT {
            id: new_id,
            owner: recipient,
            name,
            description,
            image_url
        };
        // Return both the newly minted NFT and the unchanged MintCap.
        (nft, mint_cap)
    }

    ////////////////////////////////////////////////////////////////////////////
    // NFT Transfer

        image_url: vector<u8>     // URL pointing to the NFT image.
    }

    // It is a Sui object, so it must have an id field of type UID as its first field.

        id: UID,       // Unique identifier for the minting capability.
        admin: address // The administrator authorized to mint NFTs.
    }

    ////////////////////////////////////////////////////////////////////////////

    // Creates a minting capability.
    // The sender of the transaction becomes the admin.
    // Note: In Sui, entry functions must return types with the drop ability.
    public entry fun create_mint_cap(ctx: &mut TxContext): MintCap {
