// The following Sui Move contract implements a simple NFT collection with minting,
// transfer and metadata update functionalities. The contract defines two resources:
// NFTAdmin (which grants minting authority) and NFT (which represents an individual NFT).
// Key design decisions:
// • Both NFTAdmin and NFT resources have the abilities key, store, and drop, so that they can be used as Sui objects.
// • The create_admin function creates an NFTAdmin object and assigns minting authority to the admin (the transaction sender).
// • The mint_nft function checks that only the authorized admin can mint new NFTs.
// • The transfer and update_metadata functions operate on NFT objects. In Sui, ownership is managed by the runtime,
//   so simply moving or returning an object triggers an ownership change.
// • The contract uses the transaction context (TxContext) to generate unique IDs (UID) and to obtain the sender's address.

module 0x0::my_module {
    // Import the TxContext which provides the transaction sender and UID generation functionality.
    use sui::tx_context::TxContext;
    // Import UID to serve as a unique identifier for Sui objects.
    use sui::object::UID;

    // NFTAdmin has the minting authority.
    resource struct NFTAdmin has key, store, drop {
        // For Sui objects, the first field must be a unique identifier.
        id: UID,
        // Administrator's address.
        admin: address,
    }

    // NFT represents a non-fungible token with metadata fields.
    resource struct NFT has key, store, drop {
        // Unique identifier provided by the Sui framework.
        id: UID,
        // The NFT's name.
        name: vector<u8>,
        // The NFT's description.
        description: vector<u8>,
        // URL for the NFT's image.
        image_url: vector<u8>,
    }

    // This entry function creates and returns an NFTAdmin object to grant minting authority.
    // The admin parameter is not directly used; instead, we use the sender's address from the TxContext.
    public entry fun create_admin(admin: &signer, ctx: &mut TxContext): NFTAdmin {
        let admin_addr = ctx.sender();
        NFTAdmin { id: UID::new(ctx), admin: admin_addr }
    }

    // This entry function mints a new NFT.
    // Access control is enforced by allowing only the admin (stored in NFTAdmin) to mint new NFTs.
    public entry fun mint_nft(admin_cap: &NFTAdmin, name: vector<u8>, description: vector<u8>, image_url: vector<u8>, ctx: &mut TxContext): NFT {
        // Only the admin is allowed to mint NFTs.
        assert!(ctx.sender() == admin_cap.admin, 1);
        let new_id = UID::new(ctx);
        NFT { id: new_id, name, description, image_url }
    }

    // This entry function transfers an NFT.
    // In Sui, transferring an object only requires moving the object as the transaction output.
    public entry fun transfer(nft: NFT, recipient: address, ctx: &mut TxContext): NFT {
        // The Sui runtime ensures that only the current owner can move the NFT.
        // This function simply returns the same NFT.
        nft
    }

    // This entry function updates the metadata of an NFT.
    // It creates a new NFT resource with the same UID and updated metadata.
    public entry fun update_metadata(nft: NFT, new_name: vector<u8>, new_description: vector<u8>, new_image_url: vector<u8>, ctx: &mut TxContext): NFT {
        let updated_nft = NFT { id: nft.id, name: new_name, description: new_description, image_url: new_image_url };
        updated_nft
    }
}
