module temp_addr::my_module {

    use std::signer;

    // A simple helper function to return the minimum of two u64 values.
    fun min(a: u64, b: u64): u64 {
        if (a < b) { a } else { b }
    }

    // A simple iterative algorithm to compute the integer square root of a u64.
    // This is used to initialize liquidity shares based on the geometric mean.
    public fun sqrt(n: u64): u64 {
        if (n < 2) {
            n
        } else {
            let mut x = n;
            let mut y = (x + 1) / 2;
            while (y < x) {
                x = y;
                y = (x + n / x) / 2;
            }
            x
        }
    }

    // Resource representing a liquidity pool for two tokens.
    // The pool tracks reserves for TokenA and TokenB, the fee rate (in percent),
    // accumulated fees for each token, total liquidity shares, and an admin address.
    resource struct LiquidityPool<TokenA, TokenB> has key {
        reserve_a: u64,
        reserve_b: u64,
        fee_rate: u64,    // Fee rate as a percentage (e.g., 1 means 1% fee)
        fee_a: u64,       // Accumulated fees for TokenA from swaps
        fee_b: u64,       // Accumulated fees for TokenB from swaps
        total_shares: u64, // Total liquidity shares issued
        admin: address    // Admin address allowed to collect fees
    }

    // Resource representing liquidity provider (LP) tokens.
    // These tokens grant the holder a share in the liquidity pool.
    resource struct LPToken<TokenA, TokenB> has key {
        share: u64
    }

    /////////////////////////////////////////////////////////////////////////////
    // Initialization Functions
    /////////////////////////////////////////////////////////////////////////////

    // Function to create a new liquidity pool.
    // The creator supplies initial amounts for TokenA and TokenB reserves and sets the fee rate.
    // The initial LP share is computed as the integer square root of (initial_a * initial_b).
    public fun create_pool<TokenA, TokenB>(
        admin: &signer,
        initial_a: u64,
        initial_b: u64,
        fee_rate: u64
    ): (LiquidityPool<TokenA, TokenB>, LPToken<TokenA, TokenB>) {
        // Ensure that initial liquidity amounts are positive.
        assert!(initial_a > 0, 1);
        assert!(initial_b > 0, 2);
        // Compute initial liquidity share using the geometric mean.
        let initial_shares = sqrt(initial_a * initial_b);

        let pool = LiquidityPool<TokenA, TokenB> {
            reserve_a: initial_a,
            reserve_b: initial_b,
            fee_rate: fee_rate,
            fee_a: 0,
            fee_b: 0,
            total_shares: initial_shares,
            admin: signer::address_of(admin)
        };

        let lp_token = LPToken<TokenA, TokenB> {
            share: initial_shares
        };

        (pool, lp_token)
    }

    /////////////////////////////////////////////////////////////////////////////
    // Liquidity Management Functions
    /////////////////////////////////////////////////////////////////////////////

    // Function for liquidity providers to add liquidity to the pool.
    // Liquidity must be provided in the same ratio as the current pool reserves.
    // Returns a new LPToken representing the liquidity provider's share.
    public fun add_liquidity<TokenA, TokenB>(
        pool: &mut LiquidityPool<TokenA, TokenB>,
        amount_a: u64,
        amount_b: u64
    ): LPToken<TokenA, TokenB> {
        // When the pool already has liquidity, require that the provided amounts match the pool ratio.
        if (pool.reserve_a != 0 && pool.reserve_b != 0) {
            // Enforce ratio equality: amount_a/reserve_a must equal amount_b/reserve_b.
            // To avoid division, cross-multiply and compare.
            assert!(amount_a * pool.reserve_b == amount_b * pool.reserve_a, 100);
        }

        let share: u64;
        if (pool.total_shares == 0) {
            share = sqrt(amount_a * amount_b);
        } else {
            // Calculate the share tokens to mint as the minimum of:
            // amount_a * total_shares / reserve_a and amount_b * total_shares / reserve_b.
            let share_a = amount_a * pool.total_shares / pool.reserve_a;
            let share_b = amount_b * pool.total_shares / pool.reserve_b;
            share = min(share_a, share_b);
        }

        // Update pool reserves and total shares.
        pool.reserve_a = pool.reserve_a + amount_a;
        pool.reserve_b = pool.reserve_b + amount_b;
        pool.total_shares = pool.total_shares + share;

        LPToken<TokenA, TokenB> { share }
    }

    // Function for liquidity providers to remove liquidity from the pool.
    // The provider supplies LPToken and receives TokenA and TokenB amounts proportionally.
    public fun remove_liquidity<TokenA, TokenB>(
        pool: &mut LiquidityPool<TokenA, TokenB>,
        lp: LPToken<TokenA, TokenB>
    ): (u64, u64) {
        let share = lp.share;
        // Calculate amounts to withdraw based on the share of total liquidity.
        let amount_a = share * pool.reserve_a / pool.total_shares;
        let amount_b = share * pool.reserve_b / pool.total_shares;

        // Update the pool state.
        pool.reserve_a = pool.reserve_a - amount_a;
        pool.reserve_b = pool.reserve_b - amount_b;
        pool.total_shares = pool.total_shares - share;

        // Destroy the LP token by simply consuming it (Move will drop it).
        (amount_a, amount_b)
    }

    /////////////////////////////////////////////////////////////////////////////
    // Swap Functions
    /////////////////////////////////////////////////////////////////////////////

    // Swap function to exchange TokenA for TokenB.
    // The swap uses a simple constant product formula and charges a fee.
    // The fee (as a percentage) is deducted from the input amount.
    // Returns the output amount of TokenB.
    public fun swap_a_to_b<TokenA, TokenB>(
        pool: &mut LiquidityPool<TokenA, TokenB>,
        amount_in: u64
    ): u64 {
        // Ensure that the input amount is positive.
        assert!(amount_in > 0, 200);

        // Compute fee and net input.
        let fee = amount_in * pool.fee_rate / 100;
        let net_in = amount_in - fee;

        let old_reserve_a = pool.reserve_a;
        let old_reserve_b = pool.reserve_b;

        // The invariant k = reserve_a * reserve_b is maintained (ignoring fees).
        let k = old_reserve_a * old_reserve_b;

        let new_reserve_a = old_reserve_a + net_in;
        // Ensure new_reserve_a is not zero.
        assert!(new_reserve_a > 0, 201);

        let new_reserve_b = k / new_reserve_a;
        // The output amount is the reduction in TokenB reserve.
        // Ensure that the result does not exceed the available reserve.
        assert!(old_reserve_b > new_reserve_b, 202);
        let amount_out = old_reserve_b - new_reserve_b;

        // Update pool reserves.
        pool.reserve_a = new_reserve_a;
        pool.reserve_b = new_reserve_b;
        // Accumulate the fee for TokenA.
        pool.fee_a = pool.fee_a + fee;

        amount_out
    }

    // Swap function to exchange TokenB for TokenA.
    // Uses a constant product formula with fee deduction.
    // Returns the output amount of TokenA.
    public fun swap_b_to_a<TokenA, TokenB>(
        pool: &mut LiquidityPool<TokenA, TokenB>,
        amount_in: u64
    ): u64 {
        // Ensure that the input amount is positive.
        assert!(amount_in > 0, 300);

        // Compute fee and net input.
        let fee = amount_in * pool.fee_rate / 100;
        let net_in = amount_in - fee;

        let old_reserve_b = pool.reserve_b;
        let old_reserve_a = pool.reserve_a;

        let k = old_reserve_a * old_reserve_b;

        let new_reserve_b = old_reserve_b + net_in;
        assert!(new_reserve_b > 0, 301);

        let new_reserve_a = k / new_reserve_b;
        assert!(old_reserve_a > new_reserve_a, 302);
        let amount_out = old_reserve_a - new_reserve_a;

        // Update pool reserves.
        pool.reserve_b = new_reserve_b;
        pool.reserve_a = new_reserve_a;
        // Accumulate the fee for TokenB.
        pool.fee_b = pool.fee_b + fee;

        amount_out
    }

    /////////////////////////////////////////////////////////////////////////////
    // Fee Collection Function
    /////////////////////////////////////////////////////////////////////////////

    // Admin function to collect the accumulated fees.
    // Only the admin that created the pool can call this function.
    // Returns a tuple with the collected fees for TokenA and TokenB, and resets the fee counters.
    public fun collect_fees<TokenA, TokenB>(
        pool: &mut LiquidityPool<TokenA, TokenB>,
        admin: &signer
    ): (u64, u64) {
        // Verify that the caller is the admin.
        let caller = signer::address_of(admin);
        assert!(caller == pool.admin, 400);

        let fees_a = pool.fee_a;
        let fees_b = pool.fee_b;

        // Reset the fee accumulators.
        pool.fee_a = 0;
        pool.fee_b = 0;

        (fees_a, fees_b)
    }
