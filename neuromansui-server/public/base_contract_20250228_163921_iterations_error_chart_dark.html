module temp_addr::my_module {

    // Import necessary modules for signer operations and address manipulation.
    use std::signer;
    use std::address;

    // Error code for unauthorized actions.
    const ENOTAUTHORIZED: u64 = 1;

    /////////////////////////////////////////////////////////////////////////////
    // Resource Definitions
    /////////////////////////////////////////////////////////////////////////////

    // Define a custom token resource.
    // The "MyToken" resource represents a simple fungible token with an associated amount.
    // Using "store" and "key" abilities allows the token to be stored and transferred.
    struct MyToken has store, key {
        amount: u64,
    }

    // Define an Admin capability resource.
    // Possession of this resource grants the holder the authority to mint and burn tokens.
    // The "admin" field records the address of the administrator.
    struct Admin has key, store {
        admin: address,
    }

    /////////////////////////////////////////////////////////////////////////////
    // Administrative Functions
    /////////////////////////////////////////////////////////////////////////////

    // Initialize an Admin capability.
    // This function is intended to be called once by the administrator to create an Admin object,
    // which must then be used in subsequent administrative functions (mint and burn).
    public entry fun init_admin(admin: &signer): Admin {
        Admin { admin: signer::address_of(admin) }
    }

    // Mint new tokens.
    // Only an authorized administrator (the one holding the correct Admin resource) can mint tokens.
    // This function verifies that the signer matches the stored admin address.
    // It returns both the newly minted token and the Admin capability so that the admin does not lose it.
    public entry fun mint(admin: &signer, admin_cap: Admin, amount: u64): (MyToken, Admin) {
        // Verify that the caller is the designated administrator.
        assert!(signer::address_of(admin) == admin_cap.admin, ENOTAUTHORIZED);
        let token = MyToken { amount };
        (token, admin_cap)
    }

    // Burn tokens.
    // Only an authorized administrator can burn tokens. This function consumes the token (thus "burning" it)
    // and returns the amount that was burned along with the Admin capability.
    public entry fun burn(admin: &signer, admin_cap: Admin, token: MyToken): (u64, Admin) {
        // Verify that the caller is the designated administrator.
        assert!(signer::address_of(admin) == admin_cap.admin, ENOTAUTHORIZED);
        let amount = token.amount;
        (amount, admin_cap)
    }

    /////////////////////////////////////////////////////////////////////////////
    // Token Transfer Function
    /////////////////////////////////////////////////////////////////////////////

    // Transfer tokens.
    // The owner of the token (i.e. the signer) can transfer a token to another address.
    // In Sui, returning an object from an entry function assigns its ownership to the receiver as indicated by the transaction.
    // Here, "to" represents the intended recipient's address.
    public entry fun transfer(sender: &signer, token: MyToken, to: address): MyToken {
        // In a real Sui transaction, the runtime would update the token's owner to "to" based on the transaction.
        // This function serves as a placeholder to illustrate the transfer of tokens.
        token
    }
}

// This Sui Move module defines a basic token with mint, burn, and transfer functionality.
// The module uses proper admin access control and leverages Sui primitives like UID and TxContext.
// The module is defined at the address 'temp_addr' and its name is 'my_module'.

module temp_addr::my_module {

    // Import Sui framework modules.
    use sui::object::{UID, new_uid};
    use sui::tx_context::TxContext;

    // Error code for unauthorized access.
    const ENOTAUTHORIZED: u64 = 0;

    // Resource that holds the administrator's address.
    resource struct Admin has key, store {
        admin: address,
    }

    // Custom Token resource. We assign it the abilities "key" and "store" (i.e. not "drop")
    // so that the UID field which does not have the "drop" ability can be included.
    resource struct MyToken has key, store {
        id: UID,    // Unique identifier for the token.
        value: u64, // Amount/value represented by the token.
        owner: address, // Owner's address.
    }

    // Initialize the admin. The TxContext provides the transaction sender's address.
    // The resulting Admin capability is used to restrict minting and burning actions.
    public entry fun init_admin(ctx: &mut TxContext): Admin {
        let admin_addr = TxContext::sender(ctx);
        Admin { admin: admin_addr }
    }

    // Mint new tokens. Only an authorized admin (as recorded in admin_cap) can mint tokens.
    // Returns a tuple containing the newly minted token and the admin capability.
    public entry fun mint(admin_cap: Admin, amount: u64, ctx: &mut TxContext): (MyToken, Admin) {
        let caller = TxContext::sender(ctx);
        // Ensure that the caller is the admin.
        assert(caller == admin_cap.admin, ENOTAUTHORIZED);
        // Create a new token with a unique ID and assign ownership to the caller.
        let token = MyToken {
            id: new_uid(ctx),
            value: amount,
            owner: caller,
        };
        (token, admin_cap)
    }

    // Burn tokens. Only an authorized admin can burn tokens.
